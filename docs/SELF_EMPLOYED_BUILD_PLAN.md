// Matching rules (all minimal-risk / registry-gated):
//   - The text must lead with the business marker: עסק, or a clitic prefix
//     [בלמהו]?עסק (בעסק/לעסק/מעסק/העסק/ועסק), or the deal forms עסקה/עסקת,
//     optionally preceded by a single lead-in word הוצאה/הוצאת/הכנסה/הכנסת.
//   - The token right after the marker must NOT be a number (that is the
//     numbered route's job -- we never touch it).
//   - We then greedily match the LONGEST registered name that the remainder
//     starts with (case-insensitive, after an optional separator -/—/:).
//   - ONLY names present in `list` resolve. An unknown word ("שיווק") returns
//     null, so the legacy 'עסק שיווק 200' -> null contract is preserved and the
//     message falls through to today's behaviour (bare-business / picker).
//
// Returns { n, name, rest } on a registry hit, else null.
function _resolveBusinessNamePrefix_(text, list) {
  if (!text || !Array.isArray(list) || !list.length) return null;
  var t = String(text).replace(/[\u00a0\u202f\u200b]/g, ' ').replace(/\s+/g, ' ').trim();
  if (!t) return null;

  // Strip an optional single lead-in word (expense/income) so
  // "עסק כספלה הוצאה 5 ..." and "הוצאה עסק כספלה 5 ..." both work.
  t = t.replace(/^(?:הוצאה|הוצאת|הכנסה|הכנסת)\s+/, '');

  // Require a leading business marker (standalone, clitic, or deal form) and
  // capture whatever follows it as the remainder.
  var mk = t.match(/^(?:[בלמהו]?עסק(?:ה|ת)?)\s+(.+)$/);
  if (!mk) return null;
  var afterMarker = String(mk[1] || '').trim();
  if (!afterMarker) return null;

  // Numbered route owns the "עסק <N> ..." shape -- never hijack it.
  if (/^[+-]?\d/.test(afterMarker)) return null;

  // Build a name->n index from the registry; also index biz1's name if present.
  // Longest-name-first so "כספלה סטודיו" beats "כספלה" when both are registered.
  var byNameLen = list
    .filter(function (r) { return r && r.n && r.name && String(r.name).trim(); })
    .map(function (r) { return { n: r.n, name: String(r.name).trim() }; })
    .sort(function (a, b) { return b.name.length - a.name.length; });
  if (!byNameLen.length) return null;

  var low = afterMarker.toLowerCase();
  for (var i = 0; i < byNameLen.length; i++) {
    var nm = byNameLen[i].name;
    var nmLow = nm.toLowerCase();
    // Name must appear at the START of the remainder as a whole token run,
    // followed by end / whitespace / a separator (so "כספלה" does not match
    // inside "כספלהדבר"). Allow an optional -/—/: separator after the name.
    if (low.indexOf(nmLow) !== 0) continue;
    var after = afterMarker.slice(nm.length);
    if (after.length && !/^[\s:\-—–]/.test(after)) continue;  // glued word -> not a name hit
    var rest = after.replace(/^[\s]*[-—–:]?\s*/, '').trim();
    return { n: byNameLen[i].n, name: nm, rest: rest };
  }
  return null;
}

// Load the per-owner business registry list (small wrapper so the resolver
// stays pure). Includes biz #1's persisted name so "עסק <name>" can target the
// main תנועות tab too.
function _ownerBusinessList_(ownerPhone) {
  var clean = String(ownerPhone || '').replace(/[^0-9]/g, '');
  if (!clean) return [];
  var list = [];
  try { list = kvGet('biz:owner:' + clean + ':list') || []; } catch (_e) {}
  if (!Array.isArray(list)) list = [];
  // Ensure biz #1's name is resolvable even if it predates the list index.
  var has1 = false;
  for (var i = 0; i < list.length; i++) { if (list[i] && list[i].n === 1) { has1 = true; break; } }
  if (!has1) {
    try {
      var b1 = kvGet('biz:' + clean + ':1');
      if (b1 && b1.name) list = list.concat([{ n: 1, tabName: b1.tabName || TRANSACTIONS_SHEET, name: b1.name }]);
    } catch (_e2) {}
  }
  return list;
}

## 2. The exact fix

Storage stays tenant-isolated (every key is `…:{clean}` where `clean` = sender's phone digits).

### 2a. Storage shape (additive — one new key, reusing existing ones)

```
biz:{clean}:{n}            { tabName, tabGid, n, name, aliases?, createdAt }   // EXISTING, + optional aliases[]
biz:owner:{clean}:list     [ { n, tabName, name } ]                            // EXISTING

`biz:name:{clean}` is a flat normalized-name → number map, rebuilt/extended on every numbered write and by the explicit setter. Normalization = lowercase + collapse whitespace + strip niqqud/punctuation, so "כספלה", "כספלה ", "Kesefle" each resolve. It is per-phone, so tenant A can never read tenant B's businesses.

// Normalize a business name for case/space/niqqud-insensitive matching.
// ASCII-safe: lowercases, strips Hebrew niqqud (U+0591..U+05C7), drops most
// punctuation, collapses whitespace. "Kesefle", "כספלה ", "כּספלה" -> same key.
function _normBizName_(s) {
  if (!s) return '';
  var t = String(s).trim().toLowerCase();
  // strip Hebrew points/cantillation
  t = t.replace(/[\u0591-\u05C7]/g, '');
  // drop quotes, geresh/gershayim and punctuation that users add inconsistently
  t = t.replace(/["'`.,:;!?()\[\]{}\u05F3\u05F4\u2018\u2019\u201C\u201D-]/g, ' ');
  t = t.replace(/\s+/g, ' ').trim();

// biz:name index, (2) the canonical name on each biz:{clean}:{n} record,
// (3) the per-owner list's name/tabName. Returns the integer n or null.
function _resolveBizNumberByName_(clean, name) {
  if (!clean || !name) return null;
  var key = _normBizName_(name);
  if (!key) return null;

  // (1) fast path: the learned index
  var idx = _bizNameIndex_(clean);

// Returns null if the text starts with "עסק <number>" (that is the existing
// numbered parser's job) or is not an "עסק <word...>" form.
function _parseBusinessNamePrefix_(text) {
  if (!text) return null;
  var t = String(text).trim();
  // Leave numbered commands to _parseBusinessNumberPrefix_.
  if (/^עסק\s+\d/.test(t)) return null;
  // Require "עסק " + a non-digit token (the name) to start.
  var m = t.match(/^עסק\s+(\S.*)$/);
  if (!m) return null;
  var body = String(m[1]).trim();
  // Drop a leading filler verb the user may add ("הוצאה"/"הכנסה") right after
  // the name; we re-detect income from a leading '+' in rest, as elsewhere.
  // Name = text up to a separator, an amount, or the filler word.
  var name = '', rest = '';

  // Pattern A: "name - rest" / "name: rest" / "name — rest"
  var sep = body.match(/^([^\d+][^:\-\u2014\u2013\n]{0,40}?)\s*[:\-\u2014\u2013]\s*(.+)$/);
  if (sep) {
    name = sep[1].trim();
    rest = sep[2].trim();
  } else {
    // Pattern B: "name <amount> ..." -> split at first standalone number
    // (optionally preceded by the filler verb "הוצאה"/"הכנסה" / "$" / currency).
    var b = body
      .replace(/\s+(הוצאה|הכנסה|הוצאת|הכנסת|expense|income)\s+/i, ' ');   // strip mid filler
    var am = b.match(/^(.+?)\s+([+\-]?\d.*)$/);

      name = am[1].replace(/\s+(הוצאה|הכנסה|expense|income)\s*$/i, '').trim();

      name = b.replace(/\s+(הוצאה|הכנסה|expense|income)\s*$/i, '').trim();
      rest = '';

## 1) What I found (file:line, real code)

**The gap is real and has two independent root causes.**

**(A) Named business with no number never matches.** The owner-only multi-business dispatcher (`bot/ExpenseBot_FIXED.gs:2386`) calls `_parseBusinessNumberPrefix_(__text_)`. That parser hard-requires a digit right after `עסק`:
```js

if (!m) return null;
```
So `"עסק כספלה הוצאה 5 דולר קלוד"` → no digit → returns `null` → the whole message falls through to the generic expense path, which classifies it as a plain `עסק` expense and asks for a sub-category. ```js
// lib/vat.js
// Single source of truth for Israeli VAT (מע"מ) math. As of 2026-01-01 the
// standard rate is 18% (raised from 17% on 2025-01-01). Keep this in lock-step
// with lib/invoice.js (which already encodes 18%). Historic-year reports use
// the rate that applied IN that year so a 2024 report stays at 17%.

// Standard rate by calendar year. Default (current+future) = 18.
const VAT_RATE_BY_YEAR = { 2024: 17, 2025: 17, 2026: 18 };
export const VAT_RATE_DEFAULT = 18;

// Standard rate (percent, e.g. 18) for a given calendar year.
export function vatRateForYear(year) {
  const y = parseInt(year, 10);
  if (VAT_RATE_BY_YEAR[y] != null) return VAT_RATE_BY_YEAR[y];
  return VAT_RATE_DEFAULT;
}

// The VAT component INSIDE a VAT-inclusive (gross) amount.
//   gross 118 @ 18%  -> 18.00 recoverable input VAT (תשומות).
// Returns 0 for non-positive / non-finite input. Rounds to agorot.
export function vatFromGross(gross, ratePercent = VAT_RATE_DEFAULT) {
  const g = Number(gross);
  const r = Number(ratePercent);
  if (!Number.isFinite(g) || g <= 0 || !Number.isFinite(r) || r <= 0) return 0;
  return Math.round((g * r / (100 + r)) * 100) / 100;
}

// The pre-VAT (net) base inside a gross amount. net = gross - vat.
export function netFromGross(gross, ratePercent = VAT_RATE_DEFAULT) {
  const g = Number(gross);
  if (!Number.isFinite(g) || g <= 0) return 0;
  return Math.round((g - vatFromGross(g, ratePercent)) * 100) / 100;
}

## 2. Positioning

### Israeli reality this must be grounded in (no fabricated numbers)
- **עוסק פטור** has an annual *revenue* ceiling set by law (the מחזור threshold, ~₪120K range, indexed yearly — the product must read it from config, never hardcode). A פטור does **not** charge מע"מ and issues a **קבלה**, not a חשבונית מס. Their #1 anxiety: *"am I about to cross the ceiling and get forced into עוסק מורשה?"*
- **עוסק מורשה** charges מע"מ, files דו"ח מע"מ (monthly/bi-monthly), can reclaim תשומות (input VAT), and must issue **חשבונית מס**. Their #1 chore: *separating business from personal so the רו"ח doesn't bill hours untangling it.*
- Both pay **מקדמות מס הכנסה** and **ביטוח לאומי עצמאי**, and both live or die by knowing **תזרים** (cash flow) — not P&L — because they pay tax on cash they may have already spent.
- The recognized expense buckets (שיווק, ציוד, יועצים/רו"ח, תפעול, חומרי גלם) are exactly the canonical subs already in `CATEGORY_MAP`.

**"כל הוצאה והכנסה של העסק שלך — בהודעת וואטסאפ אחת. מוכן לרו״ח, בלי אקסל, בלי אפליקציה."**
(Every business expense and income in one WhatsApp message — accountant-ready, no Excel, no app.)

### The wedge (why this wins vs the alternatives)
- **vs. Riseup / Salt:** bank-sync misses cash, ביט, foreign cards, and **business-vs-personal split** — the exact thing the owner's copy already says (`index.html:987`). An עצמאי who pays a supplier in cash or charges Anthropic on a personal card is invisible to bank-sync; Kesefle captures it because *they type it.*
- **vs. Green Invoice / iCount / Morning:** those are invoicing-first and feel like accounting software. Kesefle is **capture-first in the app they already have open 40×/day** — and (Bet 4) can hand off to Green Invoice for the actual חשבונית. Complement, not compete.
- **vs. the רו"ח's shoebox:** Kesefle's output *is* the Google Sheet the רו"ח wants, auto-categorized into recognized buckets, multi-business separated by tab. "Send your accountant one link" is the close.

This page already ranks-targets the keyword; make it convert the segment.

## 3. The 5 highest-ROI bets (prioritized)

**Bet 1 — Named-business routing that always works (`_resolveBusinessByName_`).**

Money correctness is sacred; this is a correctness bet, not a feature.

**Bet 3 — One-tap "send to my accountant" + מע"מ/תזרים summary.**
A command (e.g. "דוח רו״ח" / "מעמ") that returns a period summary: total מחזור, total הוצאות by recognized bucket, estimated מע"מ position (תשומות vs עסקאות for מורשה), and a share-link to the Sheet filtered to the business tab. **Bet 4 — Activate חשבונית export (light up `lib/invoice.js`).**
On an income message ("עסק כספלה קיבלתי 2000 מלקוח"), offer to generate a חשבונית מס/קבלה via the existing Green Invoice integration. The plumbing exists; it needs env keys, the `TODO` body-shape verification, and a פטור-vs-מורשה branch (קבלה vs חשבונית מס). Invoicing is the #1 paid job-to-be-done for עצמאים — turning capture into a billable document is the strongest reason to choose the עסק tier.

**Bet 5 — Self-employed onboarding wizard (עוסק פטור / מורשה / חברה).**
A first-run flow that asks סוג עוסק, names the business(es) → seeds the `biz:owner:list` registry (which makes Bet 1 work by construction), sets default currency, and offers the matching profession template from `lib/professions.js`. For פטור, surface the revenue-ceiling awareness (read threshold from config, never hardcode) so "you're at X% of the פטור ceiling" becomes a recurring, sticky, *uniquely-Israeli* value moment no global app can copy.

**Sequencing logic:** 1 and 2 are correctness (ship now, they fix the owner's exact bug). 3 builds the referral loop on data you already model correctly. 4 monetizes. 5 makes 1+2 self-configure and creates the פטור-ceiling moat.
